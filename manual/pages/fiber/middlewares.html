Fiber - Middlewares

As funções projetadas para fazer alterações na solicitação ou resposta são chamadas middlewares.

O método Use() registra uma rota de middleware que corresponderá a solicitações com o prefixo fornecido (que é opcional e o padrão é "/"). 


// Function signature
func (app *App) Use(args ...interface{}) Router

Para mais informações sobre Middlewares em Fiber, clique aqui

fiber.go

func main() {
    app := fiber.New()

    handler := func(c *fiber.Ctx) error {
        return c.SendStatus(fiber.StatusOK)
    }
    md1 := func(c *fiber.Ctx) error {
        c.Set("Version", "api")
        return c.Next()
    }
    md2 := func(c *fiber.Ctx) error {
        c.Set("Version", "v1")
        return c.Next()
    }

    api := app.Group("/api") // /api
    api.Use(md1) // Middleware aplicado para todas as rotas que serão criadas em /api/

    v1 := api.Group("/v1")
    v1.Use(md2) // Middleware aplicado para todas as rotas que serão criadas em /api/v1/
    v1.Get("/user", handler) // /api/v1/user

    log.Fatal(app.Listen(":3000"))
}
